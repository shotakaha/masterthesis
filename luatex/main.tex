% ここでは欧文・和文フォントのすべてをMoralerspace（のKryptonバリアント）にしています。
% ...main...: 本文フォント（明朝）
% ...sans...: 見出しフォント（ゴシック系）
% ...mono...: モノフォント（等幅系）
\usepackage{luatexja-fontspec}
% 欧文フォント
\setmainfont{ZenMaruGothic-Regular}
\setsansfont{ZenMaruGothic-Black}
\setmonofont{ZenMaruGothic-Light}
