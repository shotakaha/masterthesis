% ltjsreportはLuaLaTeXに対応した和文テンプレートです。
% jsreportからの移行する場合は、ltjsreportを利用するとよいです。
% 新規に作成する場合は、jlreqクラスを利用するとよいです。
\documentclass[draft]{ltjsreport}

% luatexjaは、LuaTeXで日本語を組版するためのパッケージです。
% どのドキュメントクラスでも、このパッケージを読み込むことで日本語を組版できます。
\usepackage{luatexja}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% パッケージの読み込み
%
% パッケージは用途別にファイルを分類してください。

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

    %source/thanks/thanks,
}
